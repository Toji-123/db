Design SQL queries for any suitable database application using SQL DML statements and perform all types of Join on the given database.

CREATE TABLE Department (
    dept_id INT PRIMARY KEY,
    dept_name VARCHAR(50)
);

CREATE TABLE Employee (
    emp_id INT PRIMARY KEY,
    emp_name VARCHAR(50),
    dept_id INT,
    FOREIGN KEY (dept_id) REFERENCES Department(dept_id)
);

INSERT INTO Department VALUES
(1, 'HR'),
(2, 'Sales'),
(3, 'IT');

INSERT INTO Employee VALUES
(101, 'Amit', 1),
(102, 'Riya', 2),
(103, 'Sumit', 3),
(104, 'John', NULL);  -- Employee with no department


1) INNER JOIN

Shows only employees that have departments.

SELECT emp_name, dept_name
FROM Employee
INNER JOIN Department
ON Employee.dept_id = Department.dept_id;

2) LEFT JOIN
Shows all employees (including those without department).

SELECT emp_name, dept_name
FROM Employee
LEFT JOIN Department
ON Employee.dept_id = Department.dept_id;

3) RIGHT JOIN

Shows all departments (even if no employees work there).

SELECT emp_name, dept_name
FROM Employee
RIGHT JOIN Department
ON Employee.dept_id = Department.dept_id;

4) FULL OUTER JOIN (If not supported â†’ use UNION)
SELECT emp_name, dept_name
FROM Employee
LEFT JOIN Department ON Employee.dept_id = Department.dept_id
UNION
SELECT emp_name, dept_name
FROM Employee
RIGHT JOIN Department ON Employee.dept_id = Department.dept_id;

5) CROSS JOIN

Makes every employee pair with every department.

SELECT emp_name, dept_name
FROM Employee
CROSS JOIN Department;


6) SELF JOIN

Compare employees with each other (example: show as friends).

SELECT e1.emp_name AS Employee, e2.emp_name AS Work_Buddy
FROM Employee e1
JOIN Employee e2
ON e1.emp_id <> e2.emp_id;

------------------------------------------------------------------------------------------
| Join Type  | What to Say                                         |
| ---------- | --------------------------------------------------- |
| INNER JOIN | Shows matching data in both tables.                 |
| LEFT JOIN  | Shows all rows from left table even if no match.    |
| RIGHT JOIN | Shows all rows from right table even if no match.   |
| FULL JOIN  | Shows all rows from both tables (match + no match). |
| CROSS JOIN | Makes all possible pairs (cartesian product).       |
| SELF JOIN  | Join table with itself to compare rows.             |
