from collections import deque

# Recursive BFS Function
def bfs(queue, graph, visited):
    if not queue:
        return
    
    node = queue.popleft()     # Remove first element from queue
    print(node, end=" ")
    visited.add(node)

    for neighbour in graph[node]:
        if neighbour not in visited:
            queue.append(neighbour)
    
    bfs(queue, graph, visited)   # Recursive Call


# Main Code
graph = {}

# Input number of vertices
n = int(input("Enter number of vertices: "))

# Taking graph input
for i in range(n):
    vertex = input(f"Enter vertex {i+1}: ")
    neighbours = input(f"Enter neighbours of {vertex} (space separated): ").split()
    graph[vertex] = neighbours

start = input("Enter starting node for BFS: ")

visited = set()
queue = deque([start])   # Initialize queue with start node

print("\nBFS Traversal:")
bfs(queue, graph, visited)
