Create a PL/SQL block that updates the balance of an account after a deposit and displays how many rows were updated using the implicit cursor attributes.


---------Assumed Table----------------
Account(acc_no, name, balance)

PL/SQL Block
SET SERVEROUTPUT ON;

DECLARE
    v_acc_no Account.acc_no%TYPE;
    v_deposit NUMBER;
BEGIN
    -- Accept input from user
    v_acc_no := &Account_Number;
    v_deposit := &Deposit_Amount;

    -- Update balance
    UPDATE Account
    SET balance = balance + v_deposit
    WHERE acc_no = v_acc_no;

    -- Display how many rows were updated
    DBMS_OUTPUT.PUT_LINE('Rows Updated: ' || SQL%ROWCOUNT);

    -- Optional: show success or failure message
    IF SQL%ROWCOUNT = 0 THEN
        DBMS_OUTPUT.PUT_LINE('No account found with given account number.');
    ELSE
        DBMS_OUTPUT.PUT_LINE('Deposit Successful!');
    END IF;

END;
/
